CREATE TABLE students(
student_id SERIAL PRIMARY KEY,
 first_name varchar(50),
 last_name varchar(50),
 module_id INT REFERENCES module(module_id)
);

CREATE TABLE module(
module_id SERIAL PRIMARY KEY,
module_name varchar(100),
);
